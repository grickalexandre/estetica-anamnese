rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ===== CLÍNICAS =====
    match /clinicas/{clinicaId} {
      // Permitir leitura para todos
      allow read: if true;
      
      // Permitir escrita para todos (desenvolvimento)
      allow write: if true;
    }
    
    // ===== USUÁRIOS =====
    match /usuarios/{userId} {
      // Permitir leitura para todos
      allow read: if true;
      
      // Permitir escrita para todos (desenvolvimento)
      allow write: if true;
    }
    
    // ===== CONFIGURAÇÕES =====
    match /configuracoes/{clinicaId} {
      // Permitir leitura para todos
      allow read: if true;
      
      // Permitir escrita para todos (desenvolvimento)
      allow write: if true;
    }
    
    // ===== ANAMNESES =====
    match /anamneses/{anamneseId} {
      // Permitir leitura para todos
      allow read: if true;
      
      // Permitir criação se clinicaId existe
      allow create: if request.resource.data.clinicaId is string
                    && request.resource.data.clinicaId.size() > 0;
      
      // Permitir atualização e exclusão
      allow update, delete: if true;
    }
    
    // ===== AGENDAMENTOS =====
    match /agendamentos/{agendamentoId} {
      allow read, write: if true;
    }
    
    // ===== ORÇAMENTOS =====
    match /orcamentos/{orcamentoId} {
      allow read, write: if true;
    }
    
    // ===== PRONTUÁRIOS =====
    match /prontuarios/{prontuarioId} {
      allow read, write: if true;
    }
    
    // ===== PLANO DE CONTAS =====
    match /plano_contas/{categoriaId} {
      allow read, write: if true;
    }
    
    // ===== CONTAS A PAGAR =====
    match /contas_pagar/{contaId} {
      allow read, write: if true;
    }
    
    // ===== CONTAS A RECEBER =====
    match /contas_receber/{contaId} {
      allow read, write: if true;
    }
    
    // ===== MOVIMENTAÇÕES =====
    match /movimentacoes/{movimentacaoId} {
      allow read, write: if true;
    }
    
    // ===== FORNECEDORES =====
    match /fornecedores/{fornecedorId} {
      allow read, write: if true;
    }
    
    // ===== PRODUTOS =====
    match /produtos/{produtoId} {
      allow read, write: if true;
    }
    
    // ===== CLIENTES =====
    match /clientes/{clienteId} {
      allow read, write: if true;
    }
    
    // ===== MOVIMENTAÇÕES ESTOQUE =====
    match /movimentacoes_estoque/{movimentacaoId} {
      allow read, write: if true;
    }
    
    // ===== CATÁLOGO PROCEDIMENTOS =====
    match /catalogo_procedimentos/{procedimentoId} {
      allow read, write: if true;
    }
    
    // ===== ATENDIMENTOS =====
    match /atendimentos/{atendimentoId} {
      allow read, write: if true;
    }
  }
}
